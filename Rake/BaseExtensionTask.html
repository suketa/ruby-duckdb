def init(name = nil, gem_spec = nil)
  @name = name
  @gem_spec = gem_spec
  @tmp_dir = 'tmp'
  @ext_dir = "ext/#{@name}"
  @lib_dir = 'lib'
  if @name and File.dirname(@name.to_s) != "."
    @lib_dir += "/#{File.dirname(@name.to_s)}"
  end
  @config_options = []
  @extra_options = ARGV.select { |i| i =~ /\A--?/ }
  @extra_sources = FileList[]
end
