def run_task(verbose)
  error = nil

  @reporter = TestTaskReporter.new(configuration)
  @reporter.run_ruby_with_valgrind do
    # RSpec::Core::RakeTask#run_task calls Kernel.exit on failure
    super
  rescue SystemExit => e
    error = e
  end

  raise error if error
end
